	k:=true;	
	p:= m + p;                                 **ERROR: array variables m and p cannot be added**
	p[10]:= a*b-20*h;                          **Not an error: array index within bounds**

	start
		declare p, u:integer;

		b:= m;

		p:= b[18]*18.56E+2;		   **ERROR: type mismatch **
